# 1. PostgreSQL 10.10 설치(소스 빌드)

  • 설치 준비
    • Source Download (opens new window)
$ dnf upgrade
  • https://www.postgresql.org/docs/10/install-procedure.html (opens new window)
  • ./configure --with-systemd
// configure 수행 중에 error 발생한 package를 추가 설치한다.
# dnf install readline-devel
# dnf install systemd-devel
# dnf install zlib-devel
  • Build
make 
su 
make install
  • DB 초기화 및 테스트
adduser postgres

chown postgres /usr/local/pgsql/data
su - postgres
/usr/local/pgsql/bin/initdb -D /usr/local/pgsql/data
/usr/local/pgsql/bin/pg_ctl -D /usr/local/pgsql/data -l logfile start
/usr/local/pgsql/bin/createdb test
/usr/local/pgsql/bin/psql test
  • .bash_profile 등록
PGHOME=/usr/local/pgsql
PGDATA=/usr/local/pgsql/data
PGPORT=5432
export PGHOME PGDATA PGPORT
export LD_LIBRARY_PATH=/usr/local/pgsql/lib:$LD_LIBRARY_PATH
export PATH=/usr/local/pgsql/bin:$PATH
  • mkdir -p /usr/local/pgsql/data/log (로그 dir 생성)
  • 설정파일 변경
    • pg_hba.conf
host    all             all             0.0.0.0/0            md5
  • postgresql.conf
listen_addresses = '*'
wal_level = logical

# yum install 로 설치한 .conf 파일과 비교하여 아래 내용 수정.
log_destination = 'stderr'
logging_collector = on
log_directory = 'log'
log_filename = 'postgresql-%a.log'
log_truncate_on_rotation = on
log_rotation_age = 1d
log_rotation_size = 0
unix_socket_directories = '/var/run/postgresql, /tmp' 

# 다음은 설정 값 테스트 필요
shared_buffers = 128MB
  • (옵션) postgresql manpage
doc/src/sgml/make install (copy to /usr/local/pgsql/share/man)
sudo vi /etc/man_db.conf
- `add:` MANDATORY_MANPATH			/usr/local/pgsql/share/man
- 혹은, export MANPATH="$(manpath -g):/usr/local/pgsql/share/man"

  • systemctl 등록

    • https://www.postgresql.org/docs/10/server-start.html (opens new window)

  • $PGDATA/postgresql.service 생성

[Unit]
Description=PostgreSQL database server
Documentation=man:postgres(1)

[Service]
Type=notify

User=postgres
Group=postgres

# Disable OOM kill on the postmaster
# 여기만 새로 추가함.(shared 메모리가 부족해서 강제로 종료되지 않도록)
OOMScoreAdjust=-1000  

ExecStart=/usr/local/pgsql/bin/postgres -D /usr/local/pgsql/data
ExecReload=/bin/kill -HUP $MAINPID
KillMode=mixed
KillSignal=SIGINT
# 복구시 시간이 오래 걸릴 수 있음(기본은 90초)
TimeoutSec=0

[Install]
WantedBy=multi-user.target
  • systemctl 서비스 등록
cd /usr/lib/systemd/system
ln -s /usr/local/pgsql/data/postgresql.service postgresql.service
systemctl enable postgresql
systemctl daemon-reload
systemctl start postgresql
systemctl stop postgresql
